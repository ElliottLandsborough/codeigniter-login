  541.     /**

  551.     /**
  552.      * Returns the tree level for the given node (assuming root node is at level 0)
  553.      * @param array $node The node to query
  554.      * @return integer The level of the supplied node
  555.      */
  556.     function getNodeLevel($node)
  557.     {
  558.         $leftcol        =       $this->left_column_name;
  559.         $rightcol       =       $this->right_column_name;
  560.         $table          =       $this->table_name;
  561.         $leftval        = (int) $node[$this->left_column_name];
  562.         $rightval       = (int) $node[$this->right_column_name];
  563.         
  564.         $sql = "SELECT      COUNT(*) AS level 
  565.                 FROM        $table  
  566.                 WHERE       $leftcol < $leftval 
  567.                 AND         $rightcol > $rightval
  568.                 ";
  569.                 
  570.         $query = $this->db->query($sql);
  571.         if($query->num_rows() > 0) {
  572.             $result = $query->result();
  573.             return $result->level;
  574.         } else {
  575.             return 0;
  576.         }
  577.     }
  578.  
  579.     /**
  580.      * Returns an array of the tree starting from the supplied node
  581.      * @param array $node The node to use as the starting point (typically root)
  582.      * @return array $tree_handle The tree represented as an array to assist with
  583.      *                             the other tree traversal operations
  584.      */
  585.     function getTreePreorder($node)
  586.     {
  587.         $table      =       $this->table_name;
  588.         $leftcol    =       $this->left_column_name;
  589.         $rightcol   =       $this->right_column_name;
  590.         $leftval    = (int) $node[$leftcol];
  591.         $rightval   = (int) $node[$rightcol];
  592.         
  593.         $sql = "SELECT      *
  594.                 FROM        $table
  595.                 WHERE       $leftcol >= $leftval
  596.                 AND         $rightcol <= $rightval
  597.                 ORDER BY    $leftcol ASC";
  598.         
  599.         $query = $this->db->query($sql);
  600.         
  601.         $treeArray = array();
  602.         
  603.         if($query->num_rows() > 0) {
  604.             foreach($query->result_array() AS $result) {
  605.                 $treeArray[] = $result;
  606.             }
  607.         }
  608.         
  609.         $retArray = array(  "result_array"  =>      $treeArray,
  610.                             "prev_left"     =>      $node[$leftcol],
  611.                             "prev_right"    =>      $node[$rightcol],
  612.                             "level"         =>      -2);
  613.         
  614.         return $retArray;
  615.     }
  616.     
  617.     /**
  618.      * Returns the next element from the tree and updates the tree_handle with the
  619.      * new positions
  620.      * @param array $tree_handle Passed by reference to allow for modifications
  621.      * @return array The next node in the tree
  622.      */
  623.     function getTreeNext(&$tree_handle)
  624.     {
  625.         $leftcol        =       $this->left_column_name;
  626.         $rightcol       =       $this->right_column_name;
  627.         
  628.         if(!empty($tree_handle['result_array'])) {
  629.             if($row = array_shift($tree_handle['result_array'])) {
  630.                 
  631.                 $tree_handle['level']+= $tree_handle['prev_left'] - $row[$leftcol] + 2;
  632.                 // store current node
  633.                 $tree_handle['prev_left']  = $row[$leftcol];
  634.                 $tree_handle['prev_right'] = $row[$rightcol];
  635.                 $tree_handle['row'] = $row;
  636.  
  637.                 return array(   $leftcol  =>  $row[$leftcol],
  638.                                 $rightcol =>  $row[$rightcol]
  639.                             );
  640.             } 
  641.         }
  642.         
  643.         return FALSE;
  644.     }
  645.  
  646.     /**
  647.      * Returns the given attribute (database field) for the current node in $tree_handle
  648.      * @param array $tree_handle The tree as an array
  649.      * @param string $attribute A string containing the fieldname to retrieve
  650.      * @return string The value requested
  651.      */
  652.     function getTreeAttribute($tree_handle,$attribute) 
  653.     {
  654.         return $tree_handle['row'][$attribute];
  655.     }
  656.     
  657.     /**
  658.      * Returns the current node of the tree contained in $tree_handle
  659.      * @param array $tree_handle The tree as an array
  660.      * @return array The left and right values of the current node
  661.      */
  662.     function getTreeCurrent($tree_handle) 
  663.     {
  664.         return array(   $this->left_column_name  =>      $tree_handle['prev_left'],
  665.                         $this->right_column_name =>      $tree_handle['prev_right']
  666.                     );
  667.     }
  668.     /**
  669.      * Returns the current level from the tree
  670.      * @param array $tree_handle The tree as an array
  671.      * @return integer The integer value of the current level
  672.      */
  673.     function getTreeLevel($tree_handle) 
  674.     {
  675.         return $tree_handle['level'];
  676.     }

  689.      * @param string $fieldname The name of the field to query
  690.      * @return string $retval The value of the field for the node looked up

  692.     function getNodeAttribute($node, $fieldname)
  693.     {

  695.         $leftcol        =       $this->left_column_name;
  696.         $rightcol       =       $this->right_column_name;
  697.         $leftval        = (int) $node[$this->left_column_name];
  698.         
  699.         $sql = "SELECT      *
  700.                 FROM        $table

  704.         
  705.

  707.             $res = $query->result();

  709.         } else {
  710.             return "";
  711.         }
  712.     }
  713.     
  714.     /**
  715.      * Renders the fields for each node starting at the given node
  716.      * @param array $node The node to start with
  717.      * @param array $fields The fields to display for each node
  718.      * @return string Sample HTML render of tree
  719.      */
  720.     function getSubTreeAsHTML($node, $fields = array()) 
  721.     {
  722.         $tree_handle = $this->getTreePreorder($node);
  723.         $retVal = "";
  724.         
  725.         while($this->getTreeNext($tree_handle)) 
  726.         {
  727.             // print indentation
  728.             $retVal .= (str_repeat("&nbsp;", $this->getTreeLevel($tree_handle)*4));
  729.     
  730.             // print requested fields
  731.             $field = reset($fields);
  732.             while($field){
  733.                 $retVal .= $tree_handle['row'][$field] . "\n";
  734.                 $field = next($fields);
  735.             }
  736.             $retVal .= "<br />\n";
  737.             
  738.         }
  739.         
  740.         return $retVal;
  741.     }

  753.     /**
  754.      * Returns a "breadcrumb" string to trace the path from the given node
  755.      * back up to the root node
  756.      * @param array $node The node to use as the starting point
  757.      * @param string $crumbfield The field to use as the text for each element of the breadcrumb trail
  758.      * @return string The generated breadcrumb string
  759.      */
  760.     function getBreadcrumbString($node, $crumbfield = "name")
  761.     {
  762.         // current node
  763.         $ret = $this->getNodeAttribute($node, $crumbfield);
  764.   
  765.         // treat ancestor nodes
  766.         while($this->getAncestor($node) != array($this->left_column_name=>0,$this->right_column_name=>0))
  767.         {
  768.             $tempNode = $this->getAncestor($node);
  769.             $ret = "" . $this->getNodeAttribute($tempNode, $crumbfield) . " &gt; " . $ret;
  770.             $node = $tempNode;
  771.         }
  772.         return $ret;
  773.     }

  784.     /**
  785.      *  _setNewNode
  786.      * 
  787.      *  Inserts a new node into the tree
  788.      *
  789.      *  @param array $node An array containing the left and right values to use
  790.      *  @param array $extrafields An associative array of field names to values for \
  791.      *                           additional columns in tree table (eg CategoryName etc)
  792.      *
  793.      *  @return boolean True/False dependent upon the success of the operation
  794.      */
  795.     private function _setNewNode($node, $extrafields)
  796.     {
  797.         $table      =       $this->table_name;
  798.         $leftcol    =       $this->left_column_name;
  799.         $rightcol   =       $this->right_column_name;
  800.         $leftval    = (int) $node[$this->left_column_name];
  801.         $rightval   = (int) $node[$this->right_column_name];
  802.         
  803.         // Handle 'othercols'
  804.         $extraFieldsArg = $extraValuesArg = "";
  805.         
  806.         if(is_array($extrafields) && !empty($extrafields)) {
  807.             $fields = array();
  808.             $values = array();
  809.             
  810.             foreach($extrafields AS $field=>$value) {
  811.                 $fields[] = $field;
  812.                 $values[] = $value;
  813.             }
  814.             $extraFieldsArg = ",`" . join("`,`",$fields) . "`";
  815.             $extraValuesArg = ",'" . join("','",$values) . "'";
  816.         }
  817.         
  818.         $sql = "INSERT INTO $table ( 
  819.                     `$leftcol`,
  820.                     `$rightcol`
  821.                      $extraFieldsArg)
  822.                  VALUES (
  823.                     '$leftval',
  824.                     '$rightval'
  825.                     $extraValuesArg
  826.                  )";
  827.                     
  828.          if($this->db->query($sql)) {
  829.              return true;
  830.          } else {
  831.              log_message('error', "Node addition failed for $left_value - $right_value");
  832.          }
  833.          
  834.          return false;
  835.     }
  836.     
  837.     
  838.     /** 
  839.      * The method that performs moving/renumbering operations
  840.      * @param array $node The node to move
  841.      * @param array $targetValue Position integer to use as the target
  842.      * @return array $newpos The new left and right values of the node moved
  843.      */
  844.     private function _moveSubtree($node, $targetValue)
  845.     { 
  846.         $sizeOfTree = $node[$this->right_column_name] - $node[$this->left_column_name] + 1;
  847.         $this->_modifyNode($targetValue, $sizeOfTree);
  848.         
  849.         
  850.         if($node[$this->left_column_name] >= $targetValue)
  851.         {
  852.             $node[$this->left_column_name] += $sizeOfTree;
  853.             $node[$this->right_column_name] += $sizeOfTree;
  854.         }
  855.   
  856.         $newpos = $this->_modifyNodeRange($node[$this->left_column_name], $node[$this->right_column_name], $targetValue - $node[$this->left_column_name]);
  857.   
  858.         $this->_modifyNode($node[$this->right_column_name]+1, - $sizeOfTree);
  859.         
  860.         if($node[$this->left_column_name] <= $targetValue)
  861.         { 
  862.             $newpos[$this->left_column_name] -= $sizeOfTree;
  863.             $newpos[$this->right_column_name] -= $sizeOfTree;
  864.         }
  865.   
  866.         return $newpos;
  867.     }
  868.  
  869.     /**
  870.      * _modifyNode
  871.      *
  872.      * Adds $changeVal to all left and right values that are greater than or
